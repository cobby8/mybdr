```ts
import { PrismaClient } from "@prisma/client";
const prisma = new PrismaClient();

async function main() {
  const match = await prisma.tournamentMatch.findUnique({
    where: { id: 124n },
    select: {
      id: true, status: true, scheduledAt: true,
      homeScore: true, awayScore: true,
      quarterScores: true,
      homeTeamId: true, awayTeamId: true,
      homeTeam: { select: { id: true, name: true } },
      awayTeam: { select: { id: true, name: true } },
    },
  });

  console.log("=== Match 124 현재 상태 ===");
  console.log(JSON.stringify(match, (_,v) => typeof v === "bigint" ? v.toString() : v, 2));

  const stats = await prisma.matchPlayerStat.findMany({
    where: { tournamentMatchId: 124n },
    select: {
      id: true, tournamentTeamPlayerId: true, points: true,
      field_goals_made: true, field_goals_attempted: true,
      three_pointers_made: true, three_pointers_attempted: true,
      free_throws_made: true, free_throws_attempted: true,
      offensive_rebounds: true, defensive_rebounds: true, total_rebounds: true,
      assists: true, steals: true, blocks: true, turnovers: true, personal_fouls: true,
      tournamentTeamPlayer: {
        select: {
          jersey_number: true,
          player: { select: { name: true } },
          tournamentTeam: { select: { id: true, name: true } },
        },
      },
    },
    orderBy: { tournamentTeamPlayer: { jersey_number: "asc" } },
  });

  console.log(`\n=== 선수 stats ${stats.length}건 ===`);
  for (const s of stats) {
    const p = s.tournamentTeamPlayer;
    console.log(`#${p?.jersey_number} ${p?.player?.name} [${p?.tournamentTeam?.name}] — PTS=${s.points} FG=${s.field_goals_made}/${s.field_goals_attempted} 3P=${s.three_pointers_made}/${s.three_pointers_attempted} FT=${s.free_throws_made}/${s.free_throws_attempted} REB=${s.total_rebounds} AST=${s.assists} STL=${s.steals} BLK=${s.blocks} TO=${s.turnovers} PF=${s.personal_fouls}`);
  }

  // 합계 검증 (OT1-cum 이어야 함 = 70/70)
  const teams = new Map<string, number>();
  for (const s of stats) {
    const teamName = s.tournamentTeamPlayer?.tournamentTeam?.name || "?";
    teams.set(teamName, (teams.get(teamName) || 0) + s.points);
  }
  console.log("\n=== 팀별 PTS 합계 (현재 DB) ===");
  for (const [team, pts] of teams) console.log(`  ${team}: ${pts}`);
  console.log(`  match.homeScore=${match?.homeScore}  awayScore=${match?.awayScore}`);
}

main().finally(() => prisma.$disconnect());
```

```ts
    console.log("\nDRY-RUN 모드. --apply 플래그 추가하여 실제 반영.");
    return;
  }

  // 6. 트랜잭션 UPDATE
  console.log("\n=== APPLY ===");
  await prisma.$transaction([
    ...updates.map(u =>
      prisma.matchPlayerStat.update({ where: { id: u.id }, data: u.data })
    ),
    prisma.tournamentMatch.update({
      where: { id: 124n },
      data: {
        homeScore: targetHome,
        awayScore: targetAway,
        quarterScores: targetQS as Prisma.InputJsonValue,
      },
    }),
  ]);
  console.log(`✅ UPDATE ${updates.length} 건 + match 1건 완료`);

  // 7. 사후 검증
  const verify = await prisma.matchPlayerStat.aggregate({
    where: { tournamentMatchId: 124n },
    _sum: { points: true },
  });
  const m2 = await prisma.tournamentMatch.findUnique({
    where: { id: 124n },
    select: { homeScore: true, awayScore: true, quarterScores: true },
  });
  console.log(`\n사후 검증:`);
  console.log(`  선수 PTS 합계: ${verify._sum.points} (기대: ${targetHome + targetAway} = ${targetHome}+${targetAway})`);
  console.log(`  match.homeScore=${m2?.homeScore}  awayScore=${m2?.awayScore}`);
  console.log(`  quarterScores=${JSON.stringify(m2?.quarterScores)}`);
}

main().catch(e => { console.error(e); process.exit(1); }).finally(() => prisma.$disconnect());
```

## 📢 STEP F — 원영 공지 (선택)

Flutter `/api/v1/*` 라우트 자체는 변경 없으나, **기록앱이 매치 124 를 재조회 시 OT2 값이 quarterScores.ot[1] 에 새로 등장**합니다. 기록앱이 ot 배열 길이 1 만 가정하고 있다면 화면 깨질 수 있음. 사전 공지 권장:

> 안녕하세요 원영님.
> mybdr.kr/live/124 매치(라이징이글스 vs 제이크루)에 OT2 기록을 수동으로 공식 반영했습니다.
> - `MatchPlayerStat` 21건 OT2-cum 값으로 UPDATE
> - `TournamentMatch.quarterScores.{home,away}.ot` = `[7, 5]` / `[7, 12]` (배열 길이 2)
> - `homeScore/awayScore` 75/82 갱신
>
> Flutter 기록앱이 ot 배열을 길이 1 만 가정하고 있으면 매치 124 표시 시 영향 있을 수 있어 공유드립니다. (`/api/v1/*` 스키마/라우트 변경 없음.)
> 향후 OT2+ 기록 가능하도록 기록앱 업데이트 필요 여부 확인 부탁드립니다.
