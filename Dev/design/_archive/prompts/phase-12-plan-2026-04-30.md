## 2. A: 시즌 통계 DB 모델 설계

### 2.1 신규 테이블 3종

```prisma
model user_season_stats {
  id                BigInt    @id @default(autoincrement())
  user_id           BigInt
  season_year       Int       // 2025, 2026...
  games_played      Int       @default(0)
  total_minutes     Int       @default(0)
  avg_rating        Decimal?  @db.Decimal(3, 2)  // 0.00~5.00
  mvp_count         Int       @default(0)
  rank_position     Int?      // 시즌 종료 시 확정
  computed_at       DateTime  @default(now()) @db.Timestamp(6)
  users             User      @relation(...)

  @@unique([user_id, season_year])
  @@index([season_year, avg_rating(sort: Desc)])
}

model shot_zone_stats {
  id                BigInt    @id @default(autoincrement())
  user_id           BigInt
  season_year       Int
  zone_code         String    @db.VarChar(20)  // "paint", "mid_left", "three_top"...
  attempts          Int       @default(0)
  makes             Int       @default(0)
  efficiency        Decimal?  @db.Decimal(4, 3) // makes/attempts
  computed_at       DateTime  @default(now()) @db.Timestamp(6)

  @@unique([user_id, season_year, zone_code])
  @@index([user_id, season_year])
}

model scouting_reports {
  id                BigInt    @id @default(autoincrement())
  subject_user_id   BigInt    // 평가받는 선수
  reporter_user_id  BigInt    // 평가자
  strengths         String    @db.Text
  weaknesses        String    @db.Text
  overall_rating    Int       // 1~5
  status            String    @default("submitted") @db.VarChar(20)
  created_at        DateTime  @default(now())
  updated_at        DateTime  @updatedAt

  @@unique([subject_user_id, reporter_user_id])
  @@index([subject_user_id, created_at(sort: Desc)])
}
```

### 2.2 집계 cron 설계

| 시점 | 작업 | 위치 |
|------|------|------|
| 매일 03:00 | 직전 day 경기 → user_season_stats 증분 갱신 | `src/app/api/cron/season-stats/route.ts` |
| 시즌 종료 시 | rank_position 확정 (avg_rating DESC) | 동일 cron + 분기 |
| 슛존 — 보류 | match_events에 zone_code 없음 → MVP 미포함 | (Phase 13+) |

shot_zone_stats는 **테이블만 선생성** + 집계 cron은 후속 phase. `match_events.event_type` 현재 2pt/3pt까지만 있어 zone 파생이 추정에 의존.

## 3. C: Portone 본인인증 데이터 모델

### 3.1 User 신규 컬럼 5종

```prisma
model User {
  // ... 기존 필드 유지
  verified_name      String?    @db.VarChar(100)   // 인증 시점 실명 스냅샷
  verified_phone     String?    @db.VarChar(20)    // 010-xxxx-xxxx
  verified_birth     DateTime?  @db.Date           // YYYY-MM-DD
  name_verified      Boolean    @default(false)    // 인증 완료 여부
  verified_at        DateTime?  @db.Timestamp(6)   // 인증 일시
}
```

### 3.2 인증 흐름

```
[1] 클라: Portone JS SDK 본인인증 위젯 호출
  └ PortOne.requestIdentityVerification({ channelKey, identityVerificationId })
[2] 콜백: imp_uid 또는 identityVerificationId 수신
[3] 서버: POST /api/web/identity/verify
  └ Portone API Secret 으로 GET /identity-verifications/{id}
  └ 응답에서 verified_name/phone/birth 추출 + zod 검증
  └ User UPDATE name_verified=true + verified_at=now()
[4] 프로필 폼: name_verified=false → "본인인증" 버튼
                name_verified=true  → readonly + verified 배지
```

### 3.3 보안 가드

| 항목 | 처리 |
|------|------|
| Portone API Secret | `.env` `PORTONE_API_SECRET` (NEXT_PUBLIC_ 금지) |
| 위변조 방지 | 클라가 보낸 verified_name 무시, 서버가 Portone API 재조회 |
| 1인 1회 | name_verified=true 면 PATCH 차단 (재인증 별도 API) |
| 익명 로그 | identity_verifications 별도 테이블 — Phase 13+ 보류 |

## 4. 통합 마이그레이션 SQL

`prisma/migrations/manual/phase_12_combined.sql` (신규):

```sql
-- ============================================
-- Phase 12: A + C 통합 마이그레이션
-- 작성: 2026-04-30
-- 적용: dev DB 먼저, 1주일 안정화 후 운영 DB
-- ============================================

BEGIN;

-- ─── A. 시즌 통계 ────────────────────────────
CREATE TABLE IF NOT EXISTS user_season_stats (
  id              BIGSERIAL PRIMARY KEY,
  user_id         BIGINT    NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  season_year     INT       NOT NULL,
  games_played    INT       NOT NULL DEFAULT 0,
  total_minutes   INT       NOT NULL DEFAULT 0,
  avg_rating      DECIMAL(3,2),
  mvp_count       INT       NOT NULL DEFAULT 0,
  rank_position   INT,
  computed_at     TIMESTAMP NOT NULL DEFAULT NOW(),
  CONSTRAINT user_season_stats_uniq UNIQUE (user_id, season_year)
);
CREATE INDEX IF NOT EXISTS idx_user_season_stats_rank
  ON user_season_stats (season_year, avg_rating DESC);

CREATE TABLE IF NOT EXISTS shot_zone_stats (
  id              BIGSERIAL PRIMARY KEY,
  user_id         BIGINT    NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  season_year     INT       NOT NULL,
  zone_code       VARCHAR(20) NOT NULL,
  attempts        INT       NOT NULL DEFAULT 0,
  makes           INT       NOT NULL DEFAULT 0,
  efficiency      DECIMAL(4,3),
  computed_at     TIMESTAMP NOT NULL DEFAULT NOW(),
  CONSTRAINT shot_zone_stats_uniq UNIQUE (user_id, season_year, zone_code)
);
CREATE INDEX IF NOT EXISTS idx_shot_zone_stats_user
  ON shot_zone_stats (user_id, season_year);

CREATE TABLE IF NOT EXISTS scouting_reports (
  id                BIGSERIAL PRIMARY KEY,
  subject_user_id   BIGINT    NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  reporter_user_id  BIGINT    NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  strengths         TEXT      NOT NULL,
  weaknesses        TEXT      NOT NULL,
  overall_rating    INT       NOT NULL CHECK (overall_rating BETWEEN 1 AND 5),
  status            VARCHAR(20) NOT NULL DEFAULT 'submitted',
  created_at        TIMESTAMP NOT NULL DEFAULT NOW(),
  updated_at        TIMESTAMP NOT NULL DEFAULT NOW(),
  CONSTRAINT scouting_reports_uniq UNIQUE (subject_user_id, reporter_user_id)
);
CREATE INDEX IF NOT EXISTS idx_scouting_reports_subject
  ON scouting_reports (subject_user_id, created_at DESC);

-- ─── C. 본인인증 컬럼 ─────────────────────────
ALTER TABLE users
  ADD COLUMN IF NOT EXISTS verified_name  VARCHAR(100),
  ADD COLUMN IF NOT EXISTS verified_phone VARCHAR(20),
  ADD COLUMN IF NOT EXISTS verified_birth DATE,
  ADD COLUMN IF NOT EXISTS name_verified  BOOLEAN NOT NULL DEFAULT FALSE,
  ADD COLUMN IF NOT EXISTS verified_at    TIMESTAMP;

CREATE INDEX IF NOT EXISTS idx_users_name_verified
  ON users (name_verified) WHERE name_verified = TRUE;

-- ─── 검증 쿼리 ──────────────────────────────
-- 신규 테이블 3개 + 컬럼 5개 확인
SELECT 'user_season_stats' AS tbl, COUNT(*) AS cnt FROM user_season_stats
UNION ALL SELECT 'shot_zone_stats',   COUNT(*) FROM shot_zone_stats
UNION ALL SELECT 'scouting_reports',  COUNT(*) FROM scouting_reports;

SELECT column_name FROM information_schema.columns
 WHERE table_name = 'users'
   AND column_name IN ('verified_name','verified_phone','verified_birth','name_verified','verified_at');

COMMIT;
```

**멱등성**: `IF NOT EXISTS` 5종 적용 — 운영 재실행 안전.
